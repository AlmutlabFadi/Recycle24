  @@index([status])
  @@index([criticality])

/// تيار الأحداث — القلب النابض للمنظومة (Event Sourcing light)
model ControlEvent {
  id                 String   @id @default(cuid())
  ts                 DateTime @default(now())
  sourceComponentKey String // FK منطقي → SystemComponent.key
  eventType          String // "AUCTION.BID_PLACED" / "LEDGER.TRANSFER_POSTED"
  severity           String   @default("info") // info | warn | critical
  tenantCountry      String? // سوريا / ألمانيا
  actorUserId        String? // المستخدم المسبب (إن وجد)
  entityType         String? // "auction" | "bid" | "ledger_txn" | "shipment"
  entityId           String? // معرف الكيان المرتبط
  traceId            String? // correlation id
  ip                 String?
  payload            Json     @default("{}")

  @@index([ts(sort: Desc)])
  @@index([eventType, ts(sort: Desc)])
  @@index([entityType, entityId])
  @@index([sourceComponentKey])
  @@index([severity])
}

/// حوادث — تجميع عدة تنبيهات/أحداث للتحقيق
model Incident {
  id          String   @id @default(cuid())
  createdAt   DateTime @default(now())
  status      String   @default("open") // open | investigating | contained | closed
  severity    String   @default("medium") // low | medium | high | critical
  title       String
  summary     String?
  ownerUserId String?
  tags        String[] @default([])
  timeline    Json     @default("[]") // نقاط زمنية مختصرة
  updatedAt   DateTime @updatedAt

  links IncidentLink[]

  @@index([status, severity])
  @@index([ownerUserId])
}

/// روابط الحوادث (ربط alert/event بحادث)

  createdAt  DateTime @default(now())

  incident Incident @relation(fields: [incidentId], references: [id], onDelete: Cascade)

  @@unique([incidentId, linkType, linkId])
  @@index([linkType, linkId])
}

/// إجراءات الاحتواء/التصعيد/الموافقة

  id                String    @id @default(cuid())

  type              String // "FREEZE_USER" | "PAUSE_AUCTION" | "BLOCK_IP"
  status            String    @default("pending") // pending | approved | rejected | executed | failed | rolled_back
  requestedByUserId String
  approvedByUserId  String?
  approvedAt        DateTime?
  executedAt        DateTime?
  reason            String // إلزامي — لماذا هذا الإجراء
  scope             Json      @default("{}") // { userId, auctionId, ip, componentKey... }
  incidentId        String?
  result            Json      @default("{}")
  updatedAt         DateTime  @updatedAt

  @@index([status, createdAt(sort: Desc)])
  @@index([type])
  @@index([requestedByUserId])
  @@index([incidentId])
}
